\begin{frame}{The Preconditioned DVS -- Algorithms}

\begin{block}{Four algorithms}

{\small{
\begin{tabular}{|c|c|c|c|}
\hline
Name & Operator & Constraint \\
\hline
\hline
DVS-BDDC &
$\displaystyle 
\Matriz{a} \, \InvSchur  \Matriz{a} \, \Schur \, \VectorD{u} =
\Matriz{a} \, \InvSchur  \VectorD{f}$ &
 $\Matriz{j} \, \VectorD{u} = 0$ \\
\hline
\hline
DVS-FETI-DP & 
$\displaystyle 
\Matriz{j} \, \Schur \,\, \Matriz{j} \, \InvSchur \, \Vector{\lambda} =
\Matriz{j} \, \Schur \,\, \Matriz{j} \, \InvSchur  \VectorD{f}$ &
 $\Matriz{a} \, \Vector{\lambda} = 0$ \\
\hline
\hline
DVS-PRIMAL & 
$\displaystyle 
\InvSchur \, \Matriz{j} \, \Schur \,\, \Matriz{j} \, \VectorD{\mathrm{v}} =
\InvSchur \, \Matriz{j} \, \Schur \,\, \Matriz{j} \, \InvSchur \, \VectorD{f}$ &
 $\Matriz{a} \, \Schur \, \VectorD{\mathrm{v}} = 0$ \\
\hline
\hline
DVS-DUAL &
$\displaystyle 
\Schur \, \Matriz{a} \, \InvSchur  \Matriz{a} \, \Vector{\mu} =
\Schur \, \Matriz{a} \, \InvSchur \, \Matriz{a} \, \Schur \,\, \Matriz{j} \, \InvSchur \, \VectorD{f}$ &
 $\Matriz{j} \, \InvSchur \Vector{\mu} = 0$ \\
\hline
\end{tabular}
}}
\end{block}

\end{frame}

\begin{frame}{DVS--BDDC}

\begin{itemize}

\item DVS-algorithms can be posed in the form:
\begin{itemize}
\item $\Matriz{M} \, \VectorD{u} = \VectorD{b}$ with restriction $\Matriz{R} \, \VectorD{u}$
\end{itemize}

\item In the case of DVS-BDDC we have:
\begin{itemize}
\item $\Matriz{M}  = \Matriz{a} \, \InvSchur  \Matriz{a} \, \Schur $
\item $\VectorD{b} = \Matriz{a} \, \InvSchur \VectorD{f}$
\item $\Matriz{R} = \Matriz{j}$
\end{itemize}

\end{itemize}

\begin{exampleblock}{DVS-BDDC in action}

\begin{columns}
\begin{column}{0.48\textwidth}
$\boxed{\Matriz{M} \, \VectorD{u} = \Matriz{a} \, \InvSchur  \Matriz{a} \, \Schur \, \VectorD{u}}$
\begin{enumerate}[<+->]
\item {\scriptsize{Apply Schur: $\VectorD{vec} \leftarrow \Schur \, \VectorD{u}$}}
\item {\scriptsize{Apply Average: $\VectorD{vec} \leftarrow \Matriz{a} \, \VectorD{vec}$}}
\item {\scriptsize{Apply Inverse of Schur: $\VectorD{vec} \leftarrow\InvSchur \VectorD{vec} $}}
\item {\scriptsize{Apply Average: $\VectorD{vec} \leftarrow \Matriz{a} \, \VectorD{vec}$}}
\end{enumerate}
\end{column}
\begin{column}{0.48\textwidth}
$\boxed{\VectorD{b} = \Matriz{a} \, \InvSchur \VectorD{f}}$
\begin{enumerate}[<+->]
\item {\scriptsize{Apply Inverse of Schur: $\VectorD{vec} \leftarrow\InvSchur \VectorD{f} $}}
\item {\scriptsize{Apply Average: $\VectorD{vec} \leftarrow \Matriz{a} \, \VectorD{vec}$}}
\end{enumerate}
\end{column}
\end{columns}
\end{exampleblock}

\end{frame}

//typedef cusp::device_memory MemorySpace;
typedef cusp::host_memory MemorySpace;
typedef float ValueType;

int csr_from_cusp(int *h_rp, int *h_cj, float *h_va, float *h_b, float *h_T, int n, int nnz)
{
  // allocate storage for (n,n) matrix with nnz nonzeros in csr format
  cusp::csr_matrix<int,ValueType,MemorySpace> A(n,n,nnz);    

  // initialize row ptr array
  for(int i = 0; i < n+1; i++) 
    A.row_offsets[i] = h_rp[i];

  // initialize column and value arrays
  for(int i = 0; i < nnz; i++) {
    A.column_indices[i] = h_cj[i];
    A.values[i] = h_va[i];
  }

  // print matrix entries
  cout << "\n ----- CSR FORMAT IN CUSP ----- \n";
  cusp::print(A);

  // allocate storage for solution (x) and right hand side (b)
  cusp::array1d<ValueType, MemorySpace> x(A.num_rows, 0);
  cusp::array1d<ValueType, MemorySpace> b(A.num_rows, 0);

  return 0;
}
