\subsection{Almacenamiento de la matriz: CSR}

\begin{minipage}{0.5\linewidth}
\begin{algorithm}[H]
	\caption{Formato CSR IJK 1D}\label{alg:CSR_IJK_1D}
	{\scriptsize
	\begin{algorithmic}[1]
		\STATE INPUT: bi, ei
		\STATE OUTPUT: $AA$, $JA$, $IA$
		\STATE I := 0
		\STATE irow := 0
		\STATE IA[irow] := I
		\FOR{$i$ := bi to ei}
		\STATE Calculate $\mbi{a}_\mbiu{W}^\mbi{i}$, $\mbi{a}_\mbiu{P}^\mbi{i}$, $\mbi{a}_\mbiu{E}^\mbi{i}$
		\STATE $AA$[I] := $\mbi{a}_\mbiu{W}^\mbi{i}$
		\STATE $JA$[I] := irow - 1
		\STATE I := I + 1
		\STATE $AA$[I] := $\mbi{a}_\mbiu{P}^\mbi{i}$
		\STATE $JA$[I] := irow
		\STATE I := I + 1
		\STATE $AA$[I] := $\mbi{a}_\mbiu{E}^\mbi{i}$
		\STATE $JA$[I] := irow + 1
		\STATE I := I+ 1
		\STATE irow := irow + 1
		\STATE IA[irow] := I  	
		\ENDFOR
		\STATE $JA$[0] := 0
		\STATE $JA$[Nx-1] := 0
	\end{algorithmic}
	}
\end{algorithm}
\end{minipage}

\subsection{Almacenamiento de la matrix: CSR}

\begin{minipage}{0.50\linewidth}
\begin{algorithm}[H]
\caption{Formato CSR IJK 2D}\label{alg:CSR_IJK_1D}
{\scriptsize
\begin{algorithmic}[1]
\STATE INPUT: bi, ei, bj, ej
\STATE OUTPUT: $AA$, $JA$, $IA$
\STATE I := 0
\STATE irow := 0
\STATE IA[irow] := I
\FOR{$i$ := bi to ei}
	\FOR{$j$ := bj to ej}
		\STATE Calculate $\mbi{a}_\mbiu{S}^\mbi{i,j}$, $\mbi{a}_\mbiu{W}^\mbi{i,j}$, $\mbi{a}_\mbiu{P}^\mbi{i,j}$, $\mbi{a}_\mbiu{E}^\mbi{i,j}$, $\mbi{a}_\mbiu{N}^\mbi{i,j}$
			\STATE $AA$[I] := $\mbi{a}_\mbiu{S}^\mbi{i,j}$
			\STATE $JA$[I] := irow - NTx
			\STATE I := I + 1
			\STATE $AA$[I] := $\mbi{a}_\mbiu{W}^\mbi{i,j}$
			\STATE $JA$[I] := irow - 1
			\STATE I := I + 1
			\STATE $AA$[I] := $\mbi{a}_\mbiu{P}^\mbi{i,j}$
			\STATE $JA$[I] := irow
			\STATE I := I + 1
			\STATE $AA$[I] := $\mbi{a}_\mbiu{E}^\mbi{i,j}$
			\STATE $JA$[I] := irow + 1
			\STATE I := I + 1
			\STATE $AA$[I] := $\mbi{a}_\mbiu{N}^\mbi{i,j}$
			\STATE $JA$[I] := irow + NTx
			\STATE I := I + 1
			\STATE irow := irow + 1
			\STATE IA[irow] := I 
	\ENDFOR
\ENDFOR
\end{algorithmic}
}
\end{algorithm}
\end{minipage}

\begin{algorithm}[H]
\caption{Corrije: $\mbi{a}_\mbiu{W}$ y $\mbi{a}_\mbiu{E}$}\label{alg:Corrige_aW_aE}
{\scriptsize
\begin{algorithmic}[1]
	\STATE OFFSET\_X = Nd * NTx
	\FOR{j := 0 to NTy-1 do}
	\STATE I := j * OFFSET\_X
	\STATE JA[I+1] := 0 
	\STATE I := I + OFFSET\_X
	\STATE JA[I-2] := 0 
	\ENDFOR
\end{algorithmic}
}
\end{algorithm}

			\STATE OFFSET\_Y := (NTx * (NTy - 1) + 1) * Nd
			\FOR{i := 0 to NTx-1 do}
			\STATE I := i * Nd
			\STATE JA[I] := 0 
			\STATE I := I + OFFSET\_Y
			\STATE JA[I-1] := 0
			\ENDFOR
		\end{algorithmic}
	}
